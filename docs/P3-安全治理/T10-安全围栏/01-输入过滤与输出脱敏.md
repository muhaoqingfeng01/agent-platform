- 责任链模式（Chain of Responsibility）

## 实现方案

### 1. 责任链架构

```
请求进入
    │
    ▼
┌─────────────────┐
│ InjectionFilter  │ → SQL 注入 / 提示词注入检测
└────┬────────────┘
     │ 通过
     ▼
┌─────────────────┐
│ JailbreakFilter  │ → 越狱提示词检测（DAN、ignore previous...）
└────┬────────────┘
     │ 通过
     ▼
┌─────────────────┐
│ SensitiveWordFilter│ → 敏感词匹配
└────┬────────────┘
     │ 通过
     ▼
┌─────────────────┐
│ LengthFilter     │ → 输入长度限制（max 10000 字符）
└────┬────────────┘
     │ 通过
     ▼
  正常编排
     │
     ▼  LLM 响应
┌─────────────────┐
│ PiiDesensitizer  │ → 身份证/手机/邮箱 → ***
└────┬────────────┘
     │
     ▼
  返回用户
```

### 2. 过滤器接口与实现

```java
public interface InputFilter {

    /** 过滤检查，抛出 SecurityBlockedException 则阻断 */
    FilterResult filter(String content, FilterContext context);

    /** 优先级，越小越先执行 */
    int order();
}

@Component
@Slf4j
public class InjectionFilter implements InputFilter {
    @Override public int order() { return 1; }

    @Override
    public FilterResult filter(String content, FilterContext context) {
        // SQL 注入特征检测
        String[] sqlPatterns = {
            "(?i)(\\b(SELECT|INSERT|UPDATE|DELETE|DROP|UNION|ALTER)\\b.*\\b(FROM|INTO|TABLE)\\b)",
            "(?i)(--|#|/\\*|\\*/)",
            "(?i)(\\bOR\\b.*=.*\\bOR\\b)"
        };
        for (String pattern : sqlPatterns) {
            if (Pattern.compile(pattern).matcher(content).find()) {
                return FilterResult.block("INJECTION", "检测到潜在的注入攻击");
            }
        }

        // 提示词注入检测
        String[] promptInjectionPatterns = {
            "(?i)(ignore\\s+(all\\s+)?(previous|above)\\s+(instructions|prompts?))",
            "(?i)(you\\s+(are|must|should|will)\\s+now\\s+(act|behave|pretend))",
            "(?i)(system\\s*prompt\\s*(:|=|is|was))",
            "(?i)(DAN\\s*(mode|prompt)?)"
        };
        for (String pattern : promptInjectionPatterns) {
            if (Pattern.compile(pattern).matcher(content).find()) {
                return FilterResult.block("JAILBREAK", "检测到越狱提示词尝试");
            }
        }

        return FilterResult.pass();
    }
}
```

### 3. PII 输出脱敏

```java
@Component
public class PiiDesensitizer {

    private static final Map<String, String> PII_PATTERNS = Map.of(
        "身份证", "(\\d{6})(\\d{8})(\\d{4})",
        "手机号", "(1[3-9]\\d)(\\d{4})(\\d{4})",
        "邮箱",   "(\\w{2})(\\w+)(@\\w+\\.\\w+)",
        "银行卡", "(\\d{4})(\\d{8,12})(\\d{4})"
    );

    public String desensitize(String text) {
        String result = text;
        result = result.replaceAll(PII_PATTERNS.get("身份证"), "$1****$3");
        result = result.replaceAll(PII_PATTERNS.get("手机号"), "$1****$3");
        result = result.replaceAll(PII_PATTERNS.get("邮箱"), "$1***$3");
        result = result.replaceAll(PII_PATTERNS.get("银行卡"), "$1****$3");
        return result;
    }

    /** 检测文本中是否包含 PII */
    public boolean containsPii(String text) {
        return PII_PATTERNS.values().stream()
                .anyMatch(p -> Pattern.compile(p).matcher(text).find());
    }
}
```

### 4. 安全事件记录

```java
@Component
public class SecurityEventRecorder {

    public void record(FilterResult result, FilterContext context) {
        SecurityEvent event = SecurityEvent.builder()
                .tenantId(TenantContext.getCurrentTenantId())
                .eventType(result.getEventType())
                .ruleId(result.getRuleId())
                .conversationId(context.getConversationId())
                .messageId(context.getMessageId())
                .originalContent(context.getOriginalContent())
                .processedContent(result.isBlocked() ? null : context.getOriginalContent())
                .matchedPattern(result.getMatchedPattern())
                .actionTaken(result.isBlocked() ? "BLOCK" : "LOG")
                .operator("SYSTEM")
                .build();
        securityEventRepository.save(event);

        if (result.isBlocked() && result.getSeverity().equals("HIGH")) {
            // 触发告警通知
            alertService.sendAlert("安全拦截: " + result.getReason());
        }
    }
}
```
