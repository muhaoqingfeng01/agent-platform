## 一、概述

每条 AI 回复下方提供点赞/点踩入口，收集用户反馈用于效果评估和持续优化。反馈数据与 P4-T12 效果评估联动，点踩自动触发 BadCase 分析工单。

**核心能力**:
- 👍👎 即时反馈切换
- 点踩弹出可选原因列表
- 反馈数据自动纳入评测数据集
- 点踩触发 BadCase 自动工单
- 反馈统计面板（P5 前端实现）

## 二、消息格式

### 2.1 反馈消息（客户端 → 服务端）

```json
{
  "type": "feedback",
  "payload": {
    "messageId": "msg_uuid_123",
    "conversationId": "conv_uuid_456",
    "rating": 1,
    "reason": "不准确",
    "comment": "回答中关于日期计算的部分有误",
    "timestamp": 1750500300000
  }
}
```

| 字段 | 类型 | 说明 |
|------|------|------|
| `messageId` | string | 被评价的消息 ID |
| `conversationId` | string | 所属会话 ID |
| `rating` | 1 \| -1 | 1=点赞, -1=点踩 |
| `reason` | string? | 点踩原因（点踩时必填） |
| `comment` | string? | 补充说明（可选） |
| `timestamp` | number | 操作时间戳 |

### 2.2 反馈提交响应

```json
{
  "code": 200,
  "message": "反馈已记录",
  "data": {
    "feedbackId": "fb_789",
    "evaluationItemCreated": true,
    "badCaseTicketId": "ticket_001"
  }
}
```

## 三、交互设计

### 3.1 反馈入口

```tsx
// src/components/FeedbackButtons/FeedbackButtons.tsx
import { useState, useCallback } from 'react';
import { Button, Modal, Radio, Input, Space, message, Tooltip } from 'antd';
import {
  LikeOutlined, LikeFilled,
  DislikeOutlined, DislikeFilled,
} from '@ant-design/icons';
import { feedbackService } from '@/services/feedback';
import { useAuthStore } from '@/stores/useAuthStore';

interface Props {
  messageId: string;
  conversationId: string;
  currentRating?: 1 | -1 | 0;  // 0=未评价
}

const DISLIKE_REASONS = [
  { key: 'inaccurate', label: '不准确', description: '回答内容包含事实性错误' },
  { key: 'incomplete', label: '不完整', description: '回答未完全解决问题' },
  { key: 'unsafe', label: '不安全', description: '回答存在安全风险或有害内容' },
  { key: 'irrelevant', label: '不相关', description: '回答与问题无关' },
  { key: 'format_issue', label: '格式问题', description: '代码、表格或排版有问题' },
  { key: 'other', label: '其他', description: '其他原因' },
];

export function FeedbackButtons({ messageId, conversationId, currentRating = 0 }: Props) {
  const [rating, setRating] = useState(currentRating);
  const [submitting, setSubmitting] = useState(false);
  const [showDislikeModal, setShowDislikeModal] = useState(false);
  const [selectedReason, setSelectedReason] = useState<string>();
  const [comment, setComment] = useState('');

  const handleLike = useCallback(async () => {
    setSubmitting(true);
    try {
      if (rating === 1) {
        // 再次点击取消点赞
        await feedbackService.cancel(conversationId, messageId);
        setRating(0);
        message.success('已取消点赞');
      } else {
        await feedbackService.submit({
          messageId,
          conversationId,
          rating: 1,
        });
        setRating(1);
        message.success('感谢你的反馈！');
      }
    } catch {
      message.error('提交失败，请重试');
    } finally {
      setSubmitting(false);
    }
  }, [messageId, conversationId, rating]);

  const handleDislike = useCallback(() => {
    if (rating === -1) {
      // 再次点击取消点踩
      feedbackService.cancel(conversationId, messageId).then(() => {
        setRating(0);
        message.success('已取消反馈');
      });
      return;
    }
    // 弹出原因选择弹窗
    setSelectedReason(undefined);
    setComment('');
    setShowDislikeModal(true);
  }, [messageId, rating]);

  const handleDislikeSubmit = useCallback(async () => {
    if (!selectedReason) {
      message.warning('请选择点踩原因');
      return;
    }

    setSubmitting(true);
    try {
      await feedbackService.submit({
        messageId,
        conversationId,
        rating: -1,
        reason: DISLIKE_REASONS.find(r => r.key === selectedReason)?.label,
        comment,
      });
      setRating(-1);
      setShowDislikeModal(false);
      message.success('感谢你的反馈，我们会持续改进！');
    } catch {
      message.error('提交失败，请重试');
    } finally {
      setSubmitting(false);
    }
  }, [messageId, conversationId, selectedReason, comment]);

  return (
    <>
      <Space size="small" className="feedback-buttons">
        <Tooltip title={rating === 1 ? '取消点赞' : '点赞'}>
          <Button
            type="text"
            size="small"
            icon={rating === 1 ? <LikeFilled style={{ color: '#1677ff' }} /> : <LikeOutlined />}
            onClick={handleLike}
            loading={submitting}
            className={rating === 1 ? 'feedback-active' : ''}
          />
        </Tooltip>

        <Tooltip title={rating === -1 ? '取消反馈' : '点踩'}>
          <Button
            type="text"
            size="small"
            icon={rating === -1 ? <DislikeFilled style={{ color: '#ff4d4f' }} /> : <DislikeOutlined />}
            onClick={handleDislike}
            className={rating === -1 ? 'feedback-active' : ''}
          />
        </Tooltip>
      </Space>

      {/* 点踩原因弹窗 */}
      <Modal
        title="帮助我们改进"
        open={showDislikeModal}
        onOk={handleDislikeSubmit}
        onCancel={() => setShowDislikeModal(false)}
        confirmLoading={submitting}
        okText="提交反馈"
        cancelText="取消"
      >
        <Space direction="vertical" style={{ width: '100%' }}>
          <Radio.Group
            value={selectedReason}
            onChange={e => setSelectedReason(e.target.value)}
          >
            <Space direction="vertical">
              {DISLIKE_REASONS.map(reason => (
                <Radio key={reason.key} value={reason.key}>
                  <strong>{reason.label}</strong>
                  <br />
                  <span style={{ color: '#8c8c8c', fontSize: 12 }}>
                    {reason.description}
                  </span>
                </Radio>
              ))}
            </Space>
          </Radio.Group>

          <Input.TextArea
            placeholder="补充说明（可选）"
            value={comment}
            onChange={e => setComment(e.target.value)}
            maxLength={500}
            showCount
            rows={3}
          />
        </Space>
      </Modal>
    </>
  );
}
```

### 3.2 状态管理

```typescript
// src/stores/useFeedbackStore.ts
import { create } from 'zustand';
import { persist } from 'zustand/middleware';

interface FeedbackState {
  // 消息反馈映射: messageId → rating
  feedbacks: Record<string, 1 | -1>;

  setFeedback: (messageId: string, rating: 1 | -1) => void;
  removeFeedback: (messageId: string) => void;
  getFeedback: (messageId: string) => 1 | -1 | 0;
}

export const useFeedbackStore = create<FeedbackState>()(
  persist(
    (set, get) => ({
      feedbacks: {},

      setFeedback: (messageId, rating) =>
        set(state => ({
          feedbacks: { ...state.feedbacks, [messageId]: rating },
        })),

      removeFeedback: (messageId) =>
        set(state => {
          const { [messageId]: _, ...rest } = state.feedbacks;
          return { feedbacks: rest };
        }),

      getFeedback: (messageId) => get().feedbacks[messageId] || 0,
    }),
    {
      name: 'agent-platform-feedback',
    }
  )
);
```

## 四、后端对接

### 4.1 API 接口

| 方法 | 路径 | 权限 | 说明 |
|------|------|------|------|
| PATCH | `/api/v1/conversations/{convId}/messages/{msgId}/feedback` | `conversation:send` | 提交/更新反馈 |
| DELETE | `/api/v1/conversations/{convId}/messages/{msgId}/feedback` | `conversation:send` | 取消反馈 |
| GET | `/api/v1/feedback/stats` | `optimization:read` | 🆕 反馈聚合统计（待实现） |

### 4.2 前端 Service 层

```typescript
// src/services/feedback.ts
import { apiClient } from './apiClient';
import { API_PREFIX } from '@/config/constants';

interface SubmitFeedbackRequest {
  messageId: string;
  conversationId: string;
  rating: 1 | -1;
  reason?: string;
  comment?: string;
}

interface SubmitFeedbackResponse {
  feedbackId: string;
  evaluationItemCreated: boolean;
  badCaseTicketId?: string;
}

interface FeedbackStats {
  totalFeedbacks: number;
  positiveRate: number;      // 点赞率
  negativeRate: number;      // 点踩率
  topReasons: Array<{ reason: string; count: number }>;
  recentTrend: Array<{ date: string; positive: number; negative: number }>;
}

export const feedbackService = {
  // 提交反馈
  submit: (data: SubmitFeedbackRequest) =>
    apiClient.patch<SubmitFeedbackResponse>(
      `${API_PREFIX}/conversations/${data.conversationId}/messages/${data.messageId}/feedback`,
      { rating: data.rating, reason: data.reason, comment: data.comment }
    ),

  // 取消反馈
  cancel: (conversationId: string, messageId: string) =>
    apiClient.delete(
      `${API_PREFIX}/conversations/${conversationId}/messages/${messageId}/feedback`
    ),

  // 🆕 反馈统计（待后端实现）
  getStats: (params?: { startDate?: string; endDate?: string }) =>
    apiClient.get<FeedbackStats>(`${API_PREFIX}/feedback/stats`, { params }),
};
```

### 4.3 后端处理链路

```
FeedbackButtons 点击
    │
    ▼
PATCH /api/v1/conversations/{convId}/messages/{msgId}/feedback
    │
    ▼
MessageController.submitFeedback()
    │
    ├─→ 更新 t_message.feedback 字段
    │
    ├─→ 发布 MessageFeedbackEvent (Spring ApplicationEvent)
    │       │
    │       ▼
    │   BadCaseAutoTicketService.onFeedbackReceived()  [@EventListener]
    │       │
    │       ├─ 如果 rating=-1: LLM 分析 BadCase 根因
    │       ├─ 创建 OptimizationTicket (PENDING)
    │       └─ 写入 t_evaluation_dataset_item (Q&A 对)
    │
    └─→ 返回 { feedbackId, evaluationItemCreated, badCaseTicketId }
```

             │
             └─→ P5 反馈统计面板 + Grafana Dashboard

## 七、配置项

```json
{
  "feedback": {
    "enableReasons": true,
    "reasons": ["不准确", "不完整", "不安全", "不相关", "格式问题", "其他"],
    "maxCommentLength": 500,
    "autoBadCaseTicket": true,
    "minDislikesForAutoTicket": 1
  }
}
```

| 配置项 | 默认值 | 说明 |
|--------|:--:|------|
| `enableReasons` | true | 点踩是否要求选择原因 |
| `maxCommentLength` | 500 | 补充说明最大字符数 |
| `autoBadCaseTicket` | true | 点踩是否自动创建 BadCase 工单 |
| `minDislikesForAutoTicket` | 1 | 同一消息累计点踩数触发自动工单 |
